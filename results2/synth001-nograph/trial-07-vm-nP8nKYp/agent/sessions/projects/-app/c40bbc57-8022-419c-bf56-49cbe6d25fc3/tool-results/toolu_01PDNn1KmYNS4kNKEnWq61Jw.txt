   160→    public void init(final StateStoreContext stateStoreContext,
   161→                     final StateStore root) {
   162→        // open the DB dir
   163→        metricsRecorder.init(metricsImpl(stateStoreContext), stateStoreContext.taskId());
   164→        openDB(stateStoreContext.appConfigs(), stateStoreContext.stateDir());
   165→
   166→        final File positionCheckpointFile = new File(stateStoreContext.stateDir(), name() + ".position");
   167→        this.positionCheckpoint = new OffsetCheckpoint(positionCheckpointFile);
   168→        this.position = StoreQueryUtils.readPositionFromCheckpoint(positionCheckpoint);
   169→
   170→        // value getter should always read directly from rocksDB
   171→        // since it is only for values that are already flushed
   172→        this.context = stateStoreContext;
   173→        stateStoreContext.register(
   174→            root,
   175→            (RecordBatchingStateRestoreCallback) this::restoreBatch,
   176→            () -> StoreQueryUtils.checkpointPosition(positionCheckpoint, position)
   177→        );
   178→        consistencyEnabled = StreamsConfig.InternalConfig.getBoolean(
   179→            stateStoreContext.appConfigs(),
   180→            IQ_CONSISTENCY_OFFSET_VECTOR_ENABLED,
   181→            false);

   184→    @SuppressWarnings("unchecked")
   185→    void openDB(final Map<String, Object> configs, final File stateDir) {
   186→        // initialize the default rocksdb options
   187→
   188→        final DBOptions dbOptions = new DBOptions();
   189→        final ColumnFamilyOptions columnFamilyOptions = new ColumnFamilyOptions();
   190→        userSpecifiedOptions = new RocksDBGenericOptionsToDbOptionsColumnFamilyOptionsAdapter(dbOptions, columnFamilyOptions);
   191→
   192→        final BlockBasedTableConfigWithAccessibleCache tableConfig = new BlockBasedTableConfigWithAccessibleCache();
   193→        cache = new LRUCache(BLOCK_CACHE_SIZE);
   194→        tableConfig.setBlockCache(cache);
   195→        tableConfig.setBlockSize(BLOCK_SIZE);
   196→
   197→        filter = new BloomFilter();
   198→        tableConfig.setFilterPolicy(filter);
   199→
   200→        userSpecifiedOptions.setOptimizeFiltersForHits(true);
   201→        userSpecifiedOptions.setTableFormatConfig(tableConfig);
   202→        userSpecifiedOptions.setWriteBufferSize(WRITE_BUFFER_SIZE);
   203→        userSpecifiedOptions.setCompressionType(COMPRESSION_TYPE);
   204→        userSpecifiedOptions.setCompactionStyle(COMPACTION_STYLE);
   205→        userSpecifiedOptions.setMaxWriteBufferNumber(MAX_WRITE_BUFFERS);
   206→        userSpecifiedOptions.setCreateIfMissing(true);
   207→        userSpecifiedOptions.setErrorIfExists(false);
   208→        userSpecifiedOptions.setInfoLogLevel(InfoLogLevel.ERROR_LEVEL);
   209→        // this is the recommended way to increase parallelism in RocksDb
   210→        // note that the current implementation of setIncreaseParallelism affects the number
   211→        // of compaction threads but not flush threads (the latter remains one). Also,
   212→        // the parallelism value needs to be at least two because of the code in
   213→        // https://github.com/facebook/rocksdb/blob/62ad0a9b19f0be4cefa70b6b32876e764b7f3c11/util/options.cc#L580
   214→        // subtracts one from the value passed to determine the number of compaction threads
   215→        // (this could be a bug in the RocksDB code and their devs have been contacted).
   216→        userSpecifiedOptions.setIncreaseParallelism(Math.max(Runtime.getRuntime().availableProcessors(), 2));
   217→
   218→        wOptions = new WriteOptions();
   219→        wOptions.setDisableWAL(true);
   220→
   221→        fOptions = new FlushOptions();
   222→        fOptions.setWaitForFlush(true);
   223→
   224→        final Class<RocksDBConfigSetter> configSetterClass =
   225→                (Class<RocksDBConfigSetter>) configs.get(StreamsConfig.ROCKSDB_CONFIG_SETTER_CLASS_CONFIG);
   226→
   227→        if (configSetterClass != null) {
   228→            configSetter = Utils.newInstance(configSetterClass);
   229→            configSetter.setConfig(name, userSpecifiedOptions, configs);
   230→        }
   231→
   232→        dbDir = new File(new File(stateDir, parentDir), name);
   233→        try {
   234→            Files.createDirectories(dbDir.getParentFile().toPath());
   235→            Files.createDirectories(dbDir.getAbsoluteFile().toPath());
   236→        } catch (final IOException fatal) {
   237→            throw new ProcessorStateException(fatal);
   238→        }
   239→
   240→        // Setup statistics before the database is opened, otherwise the statistics are not updated
   241→        // with the measurements from Rocks DB
   242→        setupStatistics(configs, dbOptions);
   243→        openRocksDB(dbOptions, columnFamilyOptions);
   244→        dbAccessor = new DirectDBAccessor(db, fOptions, wOptions);
   245→        open = true;
   246→
   247→        addValueProvidersToMetricsRecorder();
   248→    }
   249→
   250→    private void setupStatistics(final Map<String, Object> configs, final DBOptions dbOptions) {
   251→        statistics = userSpecifiedOptions.statistics();
   252→        if (statistics == null) {
   253→            if (RecordingLevel.forName((String) configs.get(METRICS_RECORDING_LEVEL_CONFIG)) == RecordingLevel.DEBUG) {
   254→                statistics = new Statistics();
   255→                dbOptions.setStatistics(statistics);
   256→            }
   257→            userSpecifiedStatistics = false;
   258→        } else {
   259→            userSpecifiedStatistics = true;
   260→        }
   261→    }
   262→
   263→    private void addValueProvidersToMetricsRecorder() {
   264→        final TableFormatConfig tableFormatConfig = userSpecifiedOptions.tableFormatConfig();
   265→        if (tableFormatConfig instanceof BlockBasedTableConfigWithAccessibleCache) {
   266→            final Cache cache = ((BlockBasedTableConfigWithAccessibleCache) tableFormatConfig).blockCache();
   267→            metricsRecorder.addValueProviders(name, db, cache, userSpecifiedStatistics ? null : statistics);
   268→        } else if (tableFormatConfig instanceof BlockBasedTableConfig) {
   269→            throw new ProcessorStateException("The used block-based table format configuration does not expose the " +
   270→                    "block cache. Use the BlockBasedTableConfig instance provided by Options#tableFormatConfig() to configure " +
   271→                    "the block-based table format of RocksDB. Do not provide a new instance of BlockBasedTableConfig to " +
   272→                    "the RocksDB options.");
   273→        } else {
   274→            metricsRecorder.addValueProviders(name, db, null, userSpecifiedStatistics ? null : statistics);
   275→        }
   276→    }

   278→    void openRocksDB(final DBOptions dbOptions,
   279→                     final ColumnFamilyOptions columnFamilyOptions) {
   280→        final List<ColumnFamilyHandle> columnFamilies = openRocksDB(
   281→                dbOptions,
   282→                new ColumnFamilyDescriptor(RocksDB.DEFAULT_COLUMN_FAMILY, columnFamilyOptions)
   283→        );
   284→
   285→        cfAccessor = new SingleColumnFamilyAccessor(columnFamilies.get(0));
   286→    }

   288→    /**
   289→     * Open RocksDB while automatically creating any requested column families that don't yet exist.
   290→     */
   291→    protected List<ColumnFamilyHandle> openRocksDB(final DBOptions dbOptions,
   292→                                                   final ColumnFamilyDescriptor defaultColumnFamilyDescriptor,
   293→                                                   final ColumnFamilyDescriptor... columnFamilyDescriptors) {
   294→        final String absolutePath = dbDir.getAbsolutePath();
   295→        final List<ColumnFamilyDescriptor> extraDescriptors = Arrays.asList(columnFamilyDescriptors);
   296→        final List<ColumnFamilyDescriptor> allDescriptors = new ArrayList<>(1 + columnFamilyDescriptors.length);
   297→        allDescriptors.add(defaultColumnFamilyDescriptor);
   298→        allDescriptors.addAll(extraDescriptors);
   299→
   300→        try {
   301→            final List<byte[]> allExisting = RocksDB.listColumnFamilies(userSpecifiedOptions, absolutePath);
   302→
   303→            // Check for unexpected column families
   304→            for (final byte[] existingFamily : allExisting) {
   305→                final boolean isExpected = allDescriptors.stream()
   306→                        .anyMatch(descriptor -> Arrays.equals(descriptor.getName(), existingFamily));
   307→                if (!isExpected) {
   308→                    if (Arrays.equals(existingFamily, RocksDBTimestampedStore.TIMESTAMPED_VALUES_COLUMN_FAMILY_NAME)) {
   309→                        throw new ProcessorStateException(
   310→                                "Store " + name + " is a timestamped key-value store and cannot be opened as a regular key-value store. " +
   311→                                "Downgrade from timestamped to regular store is not supported directly. " +
   312→                                "To downgrade, you can delete the local state in the state directory, and rebuild the store as regular key-value store from the changelog.");
   313→                    }
   314→                    if (Arrays.equals(existingFamily, RocksDBTimestampedStoreWithHeaders.TIMESTAMPED_VALUES_WITH_HEADERS_CF_NAME)) {
   315→                        final boolean openingAsTimestampedStore = allDescriptors.stream()
   316→                                .anyMatch(descriptor -> Arrays.equals(descriptor.getName(), RocksDBTimestampedStore.TIMESTAMPED_VALUES_COLUMN_FAMILY_NAME));
   317→                        if (openingAsTimestampedStore) {
   318→                            throw new ProcessorStateException(
   319→                                    "Store " + name + " is a headers-aware store and cannot be opened as a timestamped store. " +
   320→                                    "Downgrade from headers-aware to timestamped store is not supported. " +
   321→                                    "To downgrade, you can delete the local state in the state directory, and rebuild the store as timestamped store from the changelog.");
   322→                        } else {
   323→                            throw new ProcessorStateException(
   324→                                    "Store " + name + " is a headers-aware store and cannot be opened as a regular key-value store. " +
   325→                                    "Downgrade from headers-aware to regular store is not supported.");
   326→                        }
   327→                    }
   328→
   329→                    final String unexpectedFamily = new String(existingFamily, StandardCharsets.UTF_8);
   330→                    throw new ProcessorStateException(
   331→                            "Unexpected column family '" + unexpectedFamily + "' found in store " + name + ". " +
   332→                            "The store may have been created with incompatible settings.");
   333→                }
   334→            }
   335→
   336→            final List<ColumnFamilyDescriptor> existingDescriptors = new LinkedList<>();
   337→            existingDescriptors.add(defaultColumnFamilyDescriptor);
   338→            existingDescriptors.addAll(extraDescriptors.stream()
   339→                    .filter(descriptor -> allExisting.stream().anyMatch(existing -> Arrays.equals(existing, descriptor.getName())))
   340→                    .collect(Collectors.toList()));
   341→            final List<ColumnFamilyDescriptor> toCreate = extraDescriptors.stream()
   342→                    .filter(descriptor -> allExisting.stream().noneMatch(existing -> Arrays.equals(existing, descriptor.getName())))
   343→                    .collect(Collectors.toList());
   344→            final List<ColumnFamilyHandle> existingColumnFamilies = new ArrayList<>(existingDescriptors.size());
   345→            db = RocksDB.open(dbOptions, absolutePath, existingDescriptors, existingColumnFamilies);
   346→            final List<ColumnFamilyHandle> createdColumnFamilies = db.createColumnFamilies(toCreate);
   347→
   348→            return mergeColumnFamilyHandleLists(existingColumnFamilies, createdColumnFamilies, allDescriptors);
   349→
   350→        } catch (final RocksDBException e) {
   351→            throw new ProcessorStateException("Error opening store " + name + " at location " + dbDir.toString(), e);
   352→        }
   353→    }

   355→    /**
   356→     * match up the existing and created ColumnFamilyHandles with the existing/created ColumnFamilyDescriptors
   357→     * so that the order of the resultant List matches the order of the allDescriptors argument
   358→     */
   359→    private List<ColumnFamilyHandle> mergeColumnFamilyHandleLists(final List<ColumnFamilyHandle> existingColumnFamilyHandles,
   360→                                                                  final List<ColumnFamilyHandle> createdColumnFamilyHandles,
   361→                                                                  final List<ColumnFamilyDescriptor> allDescriptors) throws RocksDBException {
   362→        final List<ColumnFamilyHandle> columnFamilies = new ArrayList<>(allDescriptors.size());
   363→        int existing = 0;
   364→        int created = 0;
   365→
   366→        while (existing + created < allDescriptors.size()) {
   367→            final ColumnFamilyHandle existingHandle = existing < existingColumnFamilyHandles.size() ? existingColumnFamilyHandles.get(existing) : null;
   368→            final ColumnFamilyHandle createdHandle = created < createdColumnFamilyHandles.size() ? createdColumnFamilyHandles.get(created) : null;
   369→            if (existingHandle != null && Arrays.equals(existingHandle.getName(), allDescriptors.get(existing + created).getName())) {
   370→                columnFamilies.add(existingHandle);
   371→                existing++;
   372→            } else if (createdHandle != null && Arrays.equals(createdHandle.getName(), allDescriptors.get(existing + created).getName())) {
   373→                columnFamilies.add(createdHandle);
   374→                created++;
   375→            } else {
   376→                throw new IllegalStateException("Unable to match up column family handles with descriptors.");
   377→            }
   378→        }
   379→        return columnFamilies;
   380→    }

   411→    @Override
   412→    public synchronized void put(final Bytes key,
   413→                                 final byte[] value) {
   414→        Objects.requireNonNull(key, "key cannot be null");
   415→        validateStoreOpen();
   416→
   417→        synchronized (position) {
   418→            cfAccessor.put(dbAccessor, key.get(), value);
   419→            StoreQueryUtils.updatePosition(position, context);
   420→        }
   421→    }

   423→    @Override
   424→    public synchronized byte[] putIfAbsent(final Bytes key,
   425→                                           final byte[] value) {
   426→        Objects.requireNonNull(key, "key cannot be null");
   427→        final byte[] originalValue = get(key);
   428→        if (originalValue == null && value != null) {
   429→            put(key, value);
   430→        }
   431→        return originalValue;
   432→    }

   464→    public <PS extends Serializer<P>, P> KeyValueIterator<Bytes, byte[]> prefixScan(final P prefix,

   466→        if (!autoManagedIterators) {

   472→    <PS extends Serializer<P>, P> KeyValueIterator<Bytes, byte[]> prefixScan(final P prefix,
   473→                                                                             final PS prefixKeySerializer,
   474→                                                                             final Set<KeyValueIterator<Bytes, byte[]>> openIterators) {
   475→        if (autoManagedIterators) {
   476→            throw new IllegalStateException("Cannot specify openIterators when using auto-managed iterators");
   477→        }
   478→        return doPrefixScan(prefix, prefixKeySerializer, openIterators);
   479→    }
   480→
   481→    <PS extends Serializer<P>, P> KeyValueIterator<Bytes, byte[]> doPrefixScan(final P prefix,
   482→                                                                               final PS prefixKeySerializer,
   483→                                                                               final Set<KeyValueIterator<Bytes, byte[]>> openIterators) {
   484→        validateStoreOpen();
   485→        Objects.requireNonNull(prefix, "prefix cannot be null");
   486→        Objects.requireNonNull(prefixKeySerializer, "prefixKeySerializer cannot be null");
   487→        final Bytes prefixBytes = Bytes.wrap(prefixKeySerializer.serialize(null, prefix));
   488→
   489→        final ManagedKeyValueIterator<Bytes, byte[]> rocksDbPrefixSeekIterator = cfAccessor.prefixScan(dbAccessor, prefixBytes);
   490→        openIterators.add(rocksDbPrefixSeekIterator);
   491→        rocksDbPrefixSeekIterator.onClose(() -> openIterators.remove(rocksDbPrefixSeekIterator));
   492→
   493→        return rocksDbPrefixSeekIterator;
   494→    }

   505→    private synchronized byte[] get(final Bytes key, final Optional<ReadOptions> readOptions) {
   506→        validateStoreOpen();
   507→        try {
   508→            return readOptions.isPresent() ? cfAccessor.get(dbAccessor, key.get(), readOptions.get()) : cfAccessor.get(dbAccessor, key.get());
   509→        } catch (final RocksDBException e) {
   510→            // String format is happening in wrapping stores. So formatted message is thrown from wrapping stores.
   511→            throw new ProcessorStateException("Error while getting value for key from store " + name, e);
   512→        }
   513→    }

   515→    @Override
   516→    public synchronized byte[] delete(final Bytes key) {
   517→        Objects.requireNonNull(key, "key cannot be null");
   518→        final byte[] oldValue;
   519→        try {
   520→            oldValue = cfAccessor.getOnly(dbAccessor, key.get());
   521→        } catch (final RocksDBException e) {
   522→            // String format is happening in wrapping stores. So formatted message is thrown from wrapping stores.
   523→            throw new ProcessorStateException("Error while getting value for key from store " + name, e);
   524→        }
   525→        put(key, null);
   526→        return oldValue;
   527→    }

   529→    void deleteRange(final Bytes keyFrom, final Bytes keyTo) {
   530→        Objects.requireNonNull(keyFrom, "keyFrom cannot be null");
   531→        Objects.requireNonNull(keyTo, "keyTo cannot be null");
   532→
   533→        validateStoreOpen();
   534→
   535→        // End of key is exclusive, so we increment it by 1 byte to make keyTo inclusive.
   536→        // RocksDB's deleteRange() does not support a null upper bound so in the event
   537→        // of overflow from increment(), the operation cannot be performed and an
   538→        // IndexOutOfBoundsException will be thrown.
   539→        cfAccessor.deleteRange(dbAccessor, keyFrom.get(), Bytes.increment(keyTo).get());
   540→    }

   578→    private KeyValueIterator<Bytes, byte[]> range(final Bytes from,
   579→                                                  final Bytes to,
   580→                                                  final boolean forward,
   581→                                                  final Set<KeyValueIterator<Bytes, byte[]>> openIterators) {
   582→        if (Objects.nonNull(from) && Objects.nonNull(to) && from.compareTo(to) > 0) {
   583→            log.warn("Returning empty iterator for fetch with invalid key range: from > to. "
   584→                    + "This may be due to range arguments set in the wrong order, " +
   585→                    "or serdes that don't preserve ordering when lexicographically comparing the serialized bytes. " +
   586→                    "Note that the built-in numerical serdes do not follow this for negative numbers");
   587→            return KeyValueIterators.emptyIterator();
   588→        }
   589→
   590→        validateStoreOpen();
   591→
   592→        final ManagedKeyValueIterator<Bytes, byte[]> rocksDBRangeIterator = cfAccessor.range(dbAccessor, from, to, forward);
   593→        openIterators.add(rocksDBRangeIterator);
   594→        rocksDBRangeIterator.onClose(() -> openIterators.remove(rocksDBRangeIterator));
   595→
   596→        return rocksDBRangeIterator;
   597→    }

   638→    /**
   639→     * Return an approximate count of key-value mappings in this store.
   640→     *
   641→     * <code>RocksDB</code> cannot return an exact entry count without doing a
   642→     * full scan, so this method relies on the <code>rocksdb.estimate-num-keys</code>
   643→     * property to get an approximate count. The returned size also includes
   644→     * a count of dirty keys in the store's in-memory cache, which may lead to some
   645→     * double-counting of entries and inflate the estimate.
   646→     *
   647→     * @return an approximate count of key-value mappings in the store.
   648→     */
   649→    @Override
   650→    public long approximateNumEntries() {
   651→        validateStoreOpen();
   652→        final long numEntries;
   653→        try {
   654→            numEntries = cfAccessor.approximateNumEntries(dbAccessor);
   655→        } catch (final RocksDBException e) {
   656→            throw new ProcessorStateException("Error fetching property from store " + name, e);
   657→        }
   658→        if (isOverflowing(numEntries)) {
   659→            return Long.MAX_VALUE;
   660→        }
   661→        return numEntries;
   662→    }

   664→    private boolean isOverflowing(final long value) {
   665→        // RocksDB returns an unsigned 8-byte integer, which could overflow long
   666→        // and manifest as a negative value.
   667→        return value < 0;
   668→    }

   691→            db.write(wOptions, (WriteBatch) batch);
   692→        } else if (batch instanceof WriteBatchWithIndex) {

   702→    @Override
   703→    public synchronized void close() {
   704→        if (!open) {
   705→            return;
   706→        }
   707→
   708→        open = false;
   709→        closeOpenIterators();
   710→
   711→        if (configSetter != null) {
   712→            configSetter.close(name, userSpecifiedOptions);
   713→            configSetter = null;
   714→        }
   715→
   716→        metricsRecorder.removeValueProviders(name);
   717→
   718→        // Important: do not rearrange the order in which the below objects are closed!
   719→        // Order of closing must follow: ColumnFamilyHandle > RocksDB > DBOptions > ColumnFamilyOptions
   720→        cfAccessor.close();
   721→        dbAccessor.close();
   722→        db.close();
   723→        userSpecifiedOptions.close();
   724→        wOptions.close();
   725→        fOptions.close();
   726→        filter.close();
   727→        cache.close();
   728→        if (statistics != null) {
   729→            statistics.close();
   730→        }
   731→
   732→        cfAccessor = null;
   733→        dbAccessor = null;
   734→        userSpecifiedOptions = null;
   735→        wOptions = null;
   736→        fOptions = null;
   737→        db = null;
   738→        filter = null;
   739→        cache = null;
   740→        statistics = null;
   741→    }
   742→
   743→    private void closeOpenIterators() {
   744→        final HashSet<KeyValueIterator<Bytes, byte[]>> iterators;
   745→        synchronized (openIterators) {
   746→            iterators = new HashSet<>(openIterators);
   747→        }
   748→        if (iterators.size() != 0) {
   749→            log.warn("Closing {} open iterators for store {}", iterators.size(), name);
   750→            for (final KeyValueIterator<Bytes, byte[]> iterator : iterators) {
   751→                iterator.close();
   752→            }
   753→        }
   754→    }

  1011→    void restoreBatch(final Collection<ConsumerRecord<byte[], byte[]>> records) {
  1012→        synchronized (position) {
  1013→            try (final WriteBatch batch = new WriteBatch()) {
  1014→                for (final ConsumerRecord<byte[], byte[]> record : records) {
  1015→                    ChangelogRecordDeserializationHelper.applyChecksAndUpdatePosition(
  1016→                        record,
  1017→                        consistencyEnabled,
  1018→                        position
  1019→                    );
  1020→                    // If version headers are not present or version is V0
  1021→                    cfAccessor.addToBatch(record.key(), record.value(), batch);
  1022→                }
  1023→                write(batch);
  1024→            } catch (final RocksDBException e) {
  1025→                throw new ProcessorStateException("Error restoring batch to store " + name, e);
  1026→            }
  1027→        }
