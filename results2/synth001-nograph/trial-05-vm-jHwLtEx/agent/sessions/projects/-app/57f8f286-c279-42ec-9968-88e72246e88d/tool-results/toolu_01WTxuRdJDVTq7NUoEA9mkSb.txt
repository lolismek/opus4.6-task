   115→    /**
   116→     * Stream thread states are the possible states that a stream thread can be in.
   117→     * A thread must only be in one state at a time
   118→     * The expected state transitions with the following defined states is:
   119→     *
   120→     * <pre>
   121→     *                 +-------------+
   122→     *          +<---- | Created (0) |
   123→     *          |      +-----+-------+
   124→     *          |            |
   125→     *          |            v
   126→     *          |      +-----+-------+
   127→     *          +<---- | Starting (1)|----->+
   128→     *          |      +-----+-------+      |
   129→     *          |                           |
   130→     *          |            +<----------+  |
   131→     *          |            |           |  |
   132→     *          |            v           |  |
   133→     *          |      +-----+-------+   |  |
   134→     *          +<---- | Partitions  | --+  |
   135→     *          |      | Revoked (2) | <----+
   136→     *          |      +-----+-------+      |
   137→     *          |           |  ^            |
   138→     *          |           v  |            |
   139→     *          |      +-----+-------+      |
   140→     *          +<---- | Partitions  |      |
   141→     *          |      | Assigned (3)| <----+
   142→     *          |      +-----+-------+      |
   143→     *          |            |              |
   144→     *          |            +<----------+  |
   145→     *          |            |           |  |
   146→     *          |            v           |  |
   147→     *          |      +-----+-------+   |  |
   148→     *          |      |             | --+  |
   149→     *          |      | Running (4) | ---->+
   150→     *          |      +-----+-------+
   151→     *          |            |
   152→     *          |            v
   153→     *          |      +-----+-------+
   154→     *          +----> | Pending     |
   155→     *                 | Shutdown (5)|
   156→     *                 +-----+-------+
   157→     *                       |
   158→     *                       v
   159→     *                 +-----+-------+
   160→     *                 | Dead (6)    |
   161→     *                 +-------------+
   162→     * </pre>
   163→     *
   164→     * Note the following:
   165→     * <ul>
   166→     *     <li>Any state can go to PENDING_SHUTDOWN. That is because streams can be closed at any time.</li>
   167→     *     <li>
   168→     *         State PENDING_SHUTDOWN may want to transit to some other states other than DEAD,
   169→     *         in the corner case when the shutdown is triggered while the thread is still in the rebalance loop.
   170→     *         In this case we will forbid the transition but will not treat as an error.
   171→     *     </li>
   172→     *     <li>
   173→     *         State PARTITIONS_REVOKED may want transit to itself indefinitely, in the corner case when
   174→     *         the coordinator repeatedly fails in-between revoking partitions and assigning new partitions.
   175→     *         Also during streams instance start up PARTITIONS_REVOKED may want to transit to itself as well.
   176→     *         In this case we will allow the transition but it will be a no-op as the set of revoked partitions
   177→     *         should be empty.
   178→     *     </li>
   179→     * </ul>
   180→     */
   181→    public enum State implements ThreadStateTransitionValidator {
   182→
   183→        CREATED(1, 5),                    // 0
   184→        STARTING(2, 3, 5),                // 1
   185→        PARTITIONS_REVOKED(2, 3, 5),      // 2
   186→        PARTITIONS_ASSIGNED(2, 3, 4, 5),  // 3
   187→        RUNNING(2, 3, 4, 5),              // 4
   188→        PENDING_SHUTDOWN(6),              // 5
   189→        DEAD;                             // 6
   190→
   191→        private final Set<Integer> validTransitions = new HashSet<>();
   192→
   193→        State(final Integer... validTransitions) {
   194→            this.validTransitions.addAll(Arrays.asList(validTransitions));
   195→        }
   196→
   197→        public boolean isAlive() {
   198→            return equals(RUNNING) || equals(STARTING) || equals(PARTITIONS_REVOKED) || equals(PARTITIONS_ASSIGNED);
   199→        }
   200→
   201→        @Override
   202→        public boolean isValidTransition(final ThreadStateTransitionValidator newState) {
   203→            final State tmpState = (State) newState;
   204→            return validTransitions.contains(tmpState.ordinal());
   205→        }
   206→    }

   208→    /**
   209→     * Listen to state change events
   210→     */
   211→    public interface StateListener {
   212→
   213→        /**
   214→         * Called when state changes
   215→         *
   216→         * @param thread   thread changing state
   217→         * @param newState current state
   218→         * @param oldState previous state
   219→         */

   221→    }
   222→
   223→    /**

   226→     */

   228→        stateListener = listener;
   229→    }
   230→
   231→    public StreamThread.StateListener getStateListener() {
   232→        return stateListener;
   233→    }

   235→    /**
   236→     * @return The state this instance is in
   237→     */
   238→    public State state() {
   239→        // we do not need to use the state lock since the variable is volatile
   240→        return state;
   241→    }
   242→
   243→    void setPartitionAssignedTime(final long lastPartitionAssignedMs) {
   244→        this.lastPartitionAssignedMs = lastPartitionAssignedMs;
   245→    }
   246→
   247→    /**
   248→     * Sets the state
   249→     *
   250→     * @param newState New state
   251→     * @return The state prior to the call to setState, or null if the transition is invalid
   252→     */
   253→    State setState(final State newState) {
   254→        final State oldState;
   255→
   256→        synchronized (stateLock) {
   257→            oldState = state;
   258→
   259→            if (state == State.PENDING_SHUTDOWN && newState != State.DEAD) {
   260→                log.debug("Ignoring request to transit from PENDING_SHUTDOWN to {}: " +
   261→                              "only DEAD state is a valid next state", newState);
   262→                // when the state is already in PENDING_SHUTDOWN, all other transitions will be
   263→                // refused but we do not throw exception here
   264→                return null;
   265→            } else if (state == State.DEAD) {
   266→                log.debug("Ignoring request to transit from DEAD to {}: " +
   267→                              "no valid next state after DEAD", newState);
   268→                // when the state is already in NOT_RUNNING, all its transitions
   269→                // will be refused but we do not throw exception here
   270→                return null;
   271→            } else if (!state.isValidTransition(newState)) {
   272→                log.error("Unexpected state transition from {} to {}", oldState, newState);
   273→                throw new StreamsException(logPrefix + "Unexpected state transition from " + oldState + " to " + newState);
   274→            } else {
   275→                log.info("State transition from {} to {}", oldState, newState);
   276→            }
   277→
   278→            state = newState;
   279→            if (newState == State.RUNNING) {
   280→                updateThreadMetadata(taskManager.activeTaskMap(), taskManager.standbyTaskMap());
   281→            }
   282→
   283→            stateLock.notifyAll();
   284→        }
   285→
   286→        if (stateListener != null) {
   287→            stateListener.onChange(this, state, oldState);
   288→        }
   289→
   290→        return oldState;
   291→    }
   292→
   293→    public boolean isRunning() {
   294→        synchronized (stateLock) {
   295→            return state.isAlive();
   296→        }
   297→    }
   298→
   299→    public boolean isStartingRunningOrPartitionAssigned() {
   300→        synchronized (stateLock) {
   301→            return state.equals(State.RUNNING) || state.equals(State.STARTING) || state.equals(State.PARTITIONS_ASSIGNED);
   302→        }
   303→    }

   396→    public static StreamThread create(final TopologyMetadata topologyMetadata,

   426→
   427→        log.info("Creating restore consumer client for thread {}", threadId);
   428→        final Map<String, Object> restoreConsumerConfigs = config.getRestoreConsumerConfigs(restoreConsumerClientId(stateUpdaterId));
   429→        final Consumer<byte[], byte[]> restoreConsumer = clientSupplier.getRestoreConsumer(restoreConsumerConfigs);
   430→
   431→        final StoreChangelogReader changelogReader = new StoreChangelogReader(
   432→            time,
   433→            config,
   434→            restorationLogContext,
   435→            adminClient,
   436→            restoreConsumer,
   437→            userStateRestoreListener,
   438→            userStandbyUpdateListener
   439→        );
   440→
   441→        final ThreadCache cache = new ThreadCache(logContext, cacheSizeBytes, streamsMetrics);
   442→
   443→        final boolean proceessingThreadsEnabled = InternalConfig.processingThreadsEnabled(config.originals());
   444→        final ActiveTaskCreator activeTaskCreator = new ActiveTaskCreator(
   445→            topologyMetadata,
   446→            config,
   447→            streamsMetrics,
   448→            stateDirectory,
   449→            cache,
   450→            time,
   451→            clientSupplier,
   452→            threadId,
   453→            threadIdx,
   454→            processId,
   455→            logContext,
   456→            proceessingThreadsEnabled
   457→        );
   458→        final StandbyTaskCreator standbyTaskCreator = new StandbyTaskCreator(
   459→            topologyMetadata,
   460→            config,
   461→            streamsMetrics,
   462→            stateDirectory,
   463→            threadId,
   464→            logContext);
   465→
   466→        final Tasks tasks = new Tasks(logContext);
   467→        final boolean processingThreadsEnabled =
   468→            InternalConfig.processingThreadsEnabled(config.originals());
   469→
   470→        final DefaultTaskManager schedulingTaskManager =
   471→            maybeCreateSchedulingTaskManager(processingThreadsEnabled, topologyMetadata, time, threadId, tasks);
   472→        final StateUpdater stateUpdater =
   473→            createStateUpdater(
   474→                streamsMetrics,
   475→                config,
   476→                restoreConsumer,
   477→                changelogReader,
   478→                topologyMetadata,
   479→                time,
   480→                clientId,
   481→                threadIdx
   482→            );
   483→
   484→        final TaskManager taskManager = new TaskManager(
   485→            time,
   486→            changelogReader,
   487→            new ProcessId(processId),
   488→            logPrefix,
   489→            activeTaskCreator,
   490→            standbyTaskCreator,
   491→            tasks,
   492→            topologyMetadata,
   493→            adminClient,
   494→            stateDirectory,
   495→            stateUpdater,
   496→            schedulingTaskManager
   497→        );
   498→        referenceContainer.taskManager = taskManager;
   499→
   500→        log.info("Creating consumer client for thread {}", threadId);
   501→        final String applicationId = config.getString(StreamsConfig.APPLICATION_ID_CONFIG);
   502→        final Map<String, Object> consumerConfigs = config.getMainConsumerConfigs(applicationId, consumerClientId(threadId), threadIdx);
   503→        consumerConfigs.put(StreamsConfig.InternalConfig.REFERENCE_CONTAINER_PARTITION_ASSIGNOR, referenceContainer);
   504→
   505→        final String originalReset = (String) consumerConfigs.get(ConsumerConfig.AUTO_OFFSET_RESET_CONFIG);
   506→        // If there are any overrides, we never fall through to the consumer, but only handle offset management ourselves.

   509→        }
   510→
   511→        final MainConsumerSetup mainConsumerSetup = setupMainConsumer(topologyMetadata, config, clientSupplier, processId, consumerConfigs);
   512→
   513→        taskManager.setMainConsumer(mainConsumerSetup.mainConsumer);
   514→        referenceContainer.mainConsumer = mainConsumerSetup.mainConsumer;
   515→
   516→        final StreamsThreadMetricsDelegatingReporter reporter = new StreamsThreadMetricsDelegatingReporter(mainConsumerSetup.mainConsumer, threadId, Optional.of(stateUpdaterId));
   517→        streamsMetrics.metricsRegistry().addReporter(reporter);
   518→
   519→        // Set up topology-aware metrics caching infrastructure
   520→        final MetricsSessionManager metricsSessionManager = new MetricsSessionManager(300_000L);
   521→        final StreamsMetricsConnector metricsConnector = new StreamsMetricsConnector(metricsSessionManager);

   598→    private static Consumer<byte[], byte[]> maybeWrapConsumer(final Map<String, Object> config,
   599→                                                              final AsyncKafkaConsumer<byte[], byte[]> consumer,
   600→                                                              final Optional<StreamsRebalanceData> streamsRebalanceData) {
   601→        final Object o = config.get(InternalConfig.INTERNAL_CONSUMER_WRAPPER);
   602→        if (o == null) {
   603→            return consumer;
   604→        }
   605→
   606→        final ConsumerWrapper wrapper;
   607→        if (o instanceof String) {
   608→            try {
   609→                wrapper = Utils.newInstance((String) o, ConsumerWrapper.class);
   610→            } catch (final ClassNotFoundException e) {
   611→                throw new IllegalArgumentException(e);
   612→            }
   613→        } else if (o instanceof Class<?>) {
   614→            wrapper = (ConsumerWrapper) Utils.newInstance((Class<?>) o);
   615→        } else {
   616→            throw new IllegalArgumentException("Internal config " + InternalConfig.INTERNAL_CONSUMER_WRAPPER + " must be a class or class name");
   617→        }
   618→
   619→        wrapper.wrapConsumer(consumer, config, streamsRebalanceData);
   620→
   621→        return wrapper;
   622→    }

   700→    private static Map<String, StreamsRebalanceData.Subtopology> initBrokerTopology(final StreamsConfig config,
   701→                                                                                    final InternalTopologyBuilder internalTopologyBuilder) {
   702→        final Map<String, String> defaultTopicConfigs = new HashMap<>();
   703→        for (final Map.Entry<String, Object> entry : config.originalsWithPrefix(StreamsConfig.TOPIC_PREFIX).entrySet()) {
   704→            if (entry.getValue() != null) {
   705→                defaultTopicConfigs.put(entry.getKey(), entry.getValue().toString());
   706→            }
   707→        }
   708→        final long windowChangeLogAdditionalRetention = config.getLong(StreamsConfig.WINDOW_STORE_CHANGE_LOG_ADDITIONAL_RETENTION_MS_CONFIG);
   709→
   710→        final Map<String, StreamsRebalanceData.Subtopology> subtopologies = new HashMap<>();
   711→        final Collection<Set<String>> copartitionGroups = internalTopologyBuilder.copartitionGroups();
   712→
   713→        final Set<String> allRepartitionSourceTopics = internalTopologyBuilder.subtopologyToTopicsInfo().values().stream()
   714→            .flatMap(t -> t.repartitionSourceTopics.keySet().stream())
   715→            .collect(Collectors.toSet());
   716→
   717→        for (final Map.Entry<TopologyMetadata.Subtopology, InternalTopologyBuilder.TopicsInfo> topicsInfoEntry : internalTopologyBuilder.subtopologyToTopicsInfo()
   718→            .entrySet()) {
   719→
   720→            final HashSet<String> allSourceTopics = new HashSet<>(topicsInfoEntry.getValue().sourceTopics);
   721→            topicsInfoEntry.getValue().repartitionSourceTopics.forEach(
   722→                (repartitionSourceTopic, repartitionTopicInfo) -> {
   723→                    allSourceTopics.add(repartitionSourceTopic);
   724→                });
   725→
   726→            final Set<String> sourceTopics = topicsInfoEntry.getValue().sourceTopics.stream()
   727→                .filter(topic -> !topicsInfoEntry.getValue().repartitionSourceTopics.containsKey(topic))
   728→                .collect(Collectors.toSet());
   729→            final Set<String> repartitionSinkTopics = topicsInfoEntry.getValue().sinkTopics.stream()
   730→                .filter(allRepartitionSourceTopics::contains)
   731→                .collect(Collectors.toSet());
   732→            final Map<String, StreamsRebalanceData.TopicInfo> repartitionSourceTopics = topicsInfoEntry.getValue().repartitionSourceTopics.entrySet()
   733→                .stream()
   734→                .collect(Collectors.toMap(Map.Entry::getKey, e ->
   735→                    new StreamsRebalanceData.TopicInfo(e.getValue().numberOfPartitions(),
   736→                        Optional.of(config.getInt(StreamsConfig.REPLICATION_FACTOR_CONFIG).shortValue()),
   737→                        e.getValue().properties(defaultTopicConfigs, windowChangeLogAdditionalRetention))));
   738→            final Map<String, StreamsRebalanceData.TopicInfo> stateChangelogTopics = topicsInfoEntry.getValue().stateChangelogTopics.entrySet()
   739→                .stream()
   740→                .collect(Collectors.toMap(Map.Entry::getKey, e ->
   741→                    new StreamsRebalanceData.TopicInfo(e.getValue().numberOfPartitions(),
   742→                        Optional.of(config.getInt(StreamsConfig.REPLICATION_FACTOR_CONFIG).shortValue()),
   743→                        e.getValue().properties(defaultTopicConfigs, windowChangeLogAdditionalRetention))));
   744→
   745→            subtopologies.put(
   746→                String.valueOf(topicsInfoEntry.getKey().nodeGroupId),
   747→                new StreamsRebalanceData.Subtopology(
   748→                    sourceTopics,
   749→                    repartitionSinkTopics,
   750→                    repartitionSourceTopics,
   751→                    stateChangelogTopics,
   752→                    copartitionGroups.stream().filter(allSourceTopics::containsAll).collect(Collectors.toList())
   753→                )
   754→            );
   755→        }
   756→
   757→        if (subtopologies.values().stream().mapToInt(x -> x.copartitionGroups().size()).sum()
   758→            != copartitionGroups.size()) {
   759→            throw new IllegalStateException(
   760→                "Not all copartition groups were converted to broker topology");
   761→        }
   762→
   763→        return subtopologies;
   764→    }

   894→    /**
   895→     * Execute the stream processors
   896→     *
   897→     * @throws KafkaException   for any Kafka-related exceptions
   898→     * @throws RuntimeException for any other non-Kafka exceptions
   899→     */
   900→    @Override
   901→    public void run() {
   902→        log.info("Starting");
   903→        if (setState(State.STARTING) == null) {
   904→            log.info("StreamThread already shutdown. Not running");
   905→            return;
   906→        }
   907→        boolean cleanRun = false;
   908→        try {
   909→            taskManager.init();
   910→            initLatencyWindowsIfNeeded(time.milliseconds());
   911→            cleanRun = runLoop();
   912→        } catch (final Throwable e) {
   913→            failedStreamThreadSensor.record();
   914→            leaveGroupRequested.set(org.apache.kafka.streams.CloseOptions.GroupMembershipOperation.LEAVE_GROUP);
   915→            streamsUncaughtExceptionHandler.accept(e, false);
   916→            // Note: the above call currently rethrows the exception, so nothing below this line will be executed
   917→        } finally {
   918→            completeShutdown(cleanRun);
   919→        }
   920→    }
   921→
   922→    /**
   923→     * Main event loop for polling, and processing records through topologies.
   924→     *
   925→     * @throws IllegalStateException If store gets registered after initialized is already finished
   926→     * @throws StreamsException      if the store's change log does not contain the partition
   927→     */
   928→    boolean runLoop() {
   929→        subscribeConsumer();
   930→
   931→        // if the thread is still in the middle of a rebalance, we should keep polling
   932→        // until the rebalance is completed before we close and commit the tasks
   933→        while (isRunning() || taskManager.rebalanceInProgress()) {
   934→            try {
   935→                checkForTopologyUpdates();
   936→                // If we received the shutdown signal while waiting for a topology to be added, we can
   937→                // stop polling regardless of the rebalance status since we know there are no tasks left
   938→                if (!isRunning() && topologyMetadata.isEmpty()) {
   939→                    log.info("Shutting down thread with empty topology.");
   940→                    break;
   941→                }
   942→
   943→                maybeSendShutdown();
   944→                final long size = cacheResizeSize.getAndSet(-1L);
   945→                if (size != -1L) {
   946→                    cacheResizer.accept(size);
   947→                }
   948→                if (processingThreadsEnabled) {
   949→                    runOnceWithProcessingThreads();
   950→                } else {
   951→                    runOnceWithoutProcessingThreads();
   952→                }
   953→
   954→                maybeGetClientInstanceIds();
   955→
   956→                // Check for a scheduled rebalance but don't trigger it until the current rebalance is done
   957→                if (!taskManager.rebalanceInProgress() && nextProbingRebalanceMs.get() < time.milliseconds()) {
   958→                    log.info("Triggering the followup rebalance scheduled for {}.", Utils.toLogDateTimeFormat(nextProbingRebalanceMs.get()));
   959→                    mainConsumer.enforceRebalance("triggered followup rebalance scheduled for " + nextProbingRebalanceMs.get());
   960→                    nextProbingRebalanceMs.set(Long.MAX_VALUE);
   961→                }
   962→            } catch (final TaskCorruptedException e) {
   963→                log.warn("Detected the states of tasks " + e.corruptedTasks() + " are corrupted. " +
   964→                         "Will close the task as dirty and re-create and bootstrap from scratch.", e);
   965→                try {
   966→                    // check if any active task got corrupted. We will trigger a rebalance in that case.
   967→                    // once the task corruptions have been handled
   968→                    final boolean enforceRebalance = taskManager.handleCorruption(e.corruptedTasks());
   969→                    if (enforceRebalance && eosEnabled) {
   970→                        log.info("Active task(s) got corrupted. Triggering a rebalance.");
   971→                        mainConsumer.enforceRebalance("Active tasks corrupted");
   972→                    }
   973→                } catch (final TaskMigratedException taskMigrated) {
   974→                    handleTaskMigrated(taskMigrated);
   975→                }
   976→            } catch (final TaskMigratedException e) {
   977→                handleTaskMigrated(e);
   978→            } catch (final UnsupportedVersionException e) {
   979→                final String errorMessage = e.getMessage();
   980→                if (errorMessage != null &&
   981→                    errorMessage.startsWith("Broker unexpectedly doesn't support requireStable flag on version ")) {
   982→
   983→                    log.error("Shutting down because the Kafka cluster seems to be on a too old version. " +
   984→                              "Setting {}=\"{}\" requires broker version 2.5 or higher.",
   985→                          StreamsConfig.PROCESSING_GUARANTEE_CONFIG,
   986→                          StreamsConfig.EXACTLY_ONCE_V2);
   987→                }
   988→                failedStreamThreadSensor.record();
   989→                this.streamsUncaughtExceptionHandler.accept(new StreamsException(e), false);
   990→                return false;
   991→            } catch (final StreamsException e) {
   992→                throw e;
   993→            } catch (final Exception e) {
   994→                throw new StreamsException(e);
   995→            }
   996→        }
   997→        return true;
   998→    }
