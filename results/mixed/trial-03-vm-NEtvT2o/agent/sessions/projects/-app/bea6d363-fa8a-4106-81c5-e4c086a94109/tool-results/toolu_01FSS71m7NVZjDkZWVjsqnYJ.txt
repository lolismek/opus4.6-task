   115→    /**
   116→     * Stream thread states are the possible states that a stream thread can be in.
   117→     * A thread must only be in one state at a time
   118→     * The expected state transitions with the following defined states is:
   119→     *
   120→     * <pre>
   121→     *                 +-------------+
   122→     *          +<---- | Created (0) |
   123→     *          |      +-----+-------+
   124→     *          |            |
   125→     *          |            v
   126→     *          |      +-----+-------+
   127→     *          +<---- | Starting (1)|----->+
   128→     *          |      +-----+-------+      |
   129→     *          |                           |
   130→     *          |            +<----------+  |
   131→     *          |            |           |  |
   132→     *          |            v           |  |
   133→     *          |      +-----+-------+   |  |
   134→     *          +<---- | Partitions  | --+  |
   135→     *          |      | Revoked (2) | <----+
   136→     *          |      +-----+-------+      |
   137→     *          |           |  ^            |
   138→     *          |           v  |            |
   139→     *          |      +-----+-------+      |
   140→     *          +<---- | Partitions  |      |
   141→     *          |      | Assigned (3)| <----+
   142→     *          |      +-----+-------+      |
   143→     *          |            |              |
   144→     *          |            +<----------+  |
   145→     *          |            |           |  |
   146→     *          |            v           |  |
   147→     *          |      +-----+-------+   |  |
   148→     *          |      |             | --+  |
   149→     *          |      | Running (4) | ---->+
   150→     *          |      +-----+-------+
   151→     *          |            |
   152→     *          |            v
   153→     *          |      +-----+-------+
   154→     *          +----> | Pending     |
   155→     *                 | Shutdown (5)|
   156→     *                 +-----+-------+
   157→     *                       |
   158→     *                       v
   159→     *                 +-----+-------+
   160→     *                 | Dead (6)    |
   161→     *                 +-------------+
   162→     * </pre>
   163→     *
   164→     * Note the following:
   165→     * <ul>
   166→     *     <li>Any state can go to PENDING_SHUTDOWN. That is because streams can be closed at any time.</li>
   167→     *     <li>
   168→     *         State PENDING_SHUTDOWN may want to transit to some other states other than DEAD,
   169→     *         in the corner case when the shutdown is triggered while the thread is still in the rebalance loop.
   170→     *         In this case we will forbid the transition but will not treat as an error.
   171→     *     </li>
   172→     *     <li>
   173→     *         State PARTITIONS_REVOKED may want transit to itself indefinitely, in the corner case when
   174→     *         the coordinator repeatedly fails in-between revoking partitions and assigning new partitions.
   175→     *         Also during streams instance start up PARTITIONS_REVOKED may want to transit to itself as well.
   176→     *         In this case we will allow the transition but it will be a no-op as the set of revoked partitions
   177→     *         should be empty.
   178→     *     </li>
   179→     * </ul>
   180→     */
   181→    public enum State implements ThreadStateTransitionValidator {
   182→
   183→        CREATED(1, 5),                    // 0
   184→        STARTING(2, 3, 5),                // 1
   185→        PARTITIONS_REVOKED(2, 3, 5),      // 2
   186→        PARTITIONS_ASSIGNED(2, 3, 4, 5),  // 3
   187→        RUNNING(2, 3, 4, 5),              // 4
   188→        PENDING_SHUTDOWN(6),              // 5
   189→        DEAD;                             // 6
   190→
   191→        private final Set<Integer> validTransitions = new HashSet<>();
   192→
   193→        State(final Integer... validTransitions) {
   194→            this.validTransitions.addAll(Arrays.asList(validTransitions));
   195→        }
   196→
   197→        public boolean isAlive() {
   198→            return equals(RUNNING) || equals(STARTING) || equals(PARTITIONS_REVOKED) || equals(PARTITIONS_ASSIGNED);
   199→        }
   200→
   201→        @Override
   202→        public boolean isValidTransition(final ThreadStateTransitionValidator newState) {
   203→            final State tmpState = (State) newState;
   204→            return validTransitions.contains(tmpState.ordinal());
   205→        }
   206→    }

   208→    /**
   209→     * Listen to state change events
   210→     */
   211→    public interface StateListener {
   212→
   213→        /**
   214→         * Called when state changes
   215→         *
   216→         * @param thread   thread changing state
   217→         * @param newState current state
   218→         * @param oldState previous state
   219→         */

   221→    }
   222→
   223→    /**
   224→     * Set the {@link StreamThread.StateListener} to be notified when state changes. Note this API is internal to
   225→     * Kafka Streams and is not intended to be used by an external application.
   226→     */
   227→    public void setStateListener(final StreamThread.StateListener listener) {
   228→        stateListener = listener;
   229→    }
   230→
   231→    public StreamThread.StateListener getStateListener() {
   232→        return stateListener;
   233→    }

   235→    /**
   236→     * @return The state this instance is in
   237→     */
   238→    public State state() {
   239→        // we do not need to use the state lock since the variable is volatile
   240→        return state;
   241→    }
   242→
   243→    void setPartitionAssignedTime(final long lastPartitionAssignedMs) {
   244→        this.lastPartitionAssignedMs = lastPartitionAssignedMs;
   245→    }
   246→
   247→    /**
   248→     * Sets the state
   249→     *
   250→     * @param newState New state
   251→     * @return The state prior to the call to setState, or null if the transition is invalid
   252→     */
   253→    State setState(final State newState) {
   254→        final State oldState;
   255→
   256→        synchronized (stateLock) {
   257→            oldState = state;
   258→
   259→            if (state == State.PENDING_SHUTDOWN && newState != State.DEAD) {
   260→                log.debug("Ignoring request to transit from PENDING_SHUTDOWN to {}: " +
   261→                              "only DEAD state is a valid next state", newState);
   262→                // when the state is already in PENDING_SHUTDOWN, all other transitions will be
   263→                // refused but we do not throw exception here
   264→                return null;
   265→            } else if (state == State.DEAD) {
   266→                log.debug("Ignoring request to transit from DEAD to {}: " +
   267→                              "no valid next state after DEAD", newState);
   268→                // when the state is already in NOT_RUNNING, all its transitions
   269→                // will be refused but we do not throw exception here
   270→                return null;
   271→            } else if (!state.isValidTransition(newState)) {
   272→                log.error("Unexpected state transition from {} to {}", oldState, newState);
   273→                throw new StreamsException(logPrefix + "Unexpected state transition from " + oldState + " to " + newState);
   274→            } else {
   275→                log.info("State transition from {} to {}", oldState, newState);
   276→            }
   277→
   278→            state = newState;
   279→            if (newState == State.RUNNING) {
   280→                updateThreadMetadata(taskManager.activeTaskMap(), taskManager.standbyTaskMap());
   281→            }
   282→
   283→            stateLock.notifyAll();
   284→        }
   285→
   286→        if (stateListener != null) {
   287→            stateListener.onChange(this, state, oldState);
   288→        }
   289→
   290→        return oldState;
   291→    }
   292→
   293→    public boolean isRunning() {
   294→        synchronized (stateLock) {
   295→            return state.isAlive();
   296→        }
   297→    }
   298→
   299→    public boolean isStartingRunningOrPartitionAssigned() {
   300→        synchronized (stateLock) {
   301→            return state.equals(State.RUNNING) || state.equals(State.STARTING) || state.equals(State.PARTITIONS_ASSIGNED);
   302→        }
   303→    }

   455→            logContext,

   458→        final StandbyTaskCreator standbyTaskCreator = new StandbyTaskCreator(
   459→            topologyMetadata,
   460→            config,
   461→            streamsMetrics,
   462→            stateDirectory,
   463→            threadId,
   464→            logContext);
   465→
   466→        final Tasks tasks = new Tasks(logContext);
   467→        final boolean processingThreadsEnabled =
   468→            InternalConfig.processingThreadsEnabled(config.originals());
   469→
   470→        final DefaultTaskManager schedulingTaskManager =
   471→            maybeCreateSchedulingTaskManager(processingThreadsEnabled, topologyMetadata, time, threadId, tasks);
   472→        final StateUpdater stateUpdater =
   473→            createStateUpdater(
   474→                streamsMetrics,
   475→                config,
   476→                restoreConsumer,
   477→                changelogReader,
   478→                topologyMetadata,
   479→                time,
   480→                clientId,
   481→                threadIdx
   482→            );
   483→
   484→        final TaskManager taskManager = new TaskManager(
   485→            time,
   486→            changelogReader,
   487→            new ProcessId(processId),
   488→            logPrefix,
   489→            activeTaskCreator,
   490→            standbyTaskCreator,
   491→            tasks,
   492→            topologyMetadata,
   493→            adminClient,
   494→            stateDirectory,

   496→            schedulingTaskManager
   497→        );
   498→        referenceContainer.taskManager = taskManager;
   499→

   501→        final String applicationId = config.getString(StreamsConfig.APPLICATION_ID_CONFIG);

   518→
   519→        final StreamThread streamThread = new StreamThread(
   520→            time,
   521→            config,
   522→            adminClient,
   523→            mainConsumerSetup.mainConsumer,
   524→            restoreConsumer,
   525→            changelogReader,
   526→            originalReset,
   527→            taskManager,
   528→            stateUpdater,

   588→    private static Consumer<byte[], byte[]> maybeWrapConsumer(final Map<String, Object> config,
   589→                                                              final AsyncKafkaConsumer<byte[], byte[]> consumer,
   590→                                                              final Optional<StreamsRebalanceData> streamsRebalanceData) {
   591→        final Object o = config.get(InternalConfig.INTERNAL_CONSUMER_WRAPPER);
   592→        if (o == null) {
   593→            return consumer;
   594→        }
   595→
   596→        final ConsumerWrapper wrapper;
   597→        if (o instanceof String) {
   598→            try {
   599→                wrapper = Utils.newInstance((String) o, ConsumerWrapper.class);
   600→            } catch (final ClassNotFoundException e) {
   601→                throw new IllegalArgumentException(e);
   602→            }
   603→        } else if (o instanceof Class<?>) {
   604→            wrapper = (ConsumerWrapper) Utils.newInstance((Class<?>) o);
   605→        } else {
   606→            throw new IllegalArgumentException("Internal config " + InternalConfig.INTERNAL_CONSUMER_WRAPPER + " must be a class or class name");
   607→        }
   608→
   609→        wrapper.wrapConsumer(consumer, config, streamsRebalanceData);
   610→
   611→        return wrapper;
   612→    }

   691→    private static Map<String, StreamsRebalanceData.Subtopology> initBrokerTopology(final StreamsConfig config,
   692→                                                                                    final InternalTopologyBuilder internalTopologyBuilder) {
   693→        final Map<String, String> defaultTopicConfigs = new HashMap<>();
   694→        for (final Map.Entry<String, Object> entry : config.originalsWithPrefix(StreamsConfig.TOPIC_PREFIX).entrySet()) {
   695→            if (entry.getValue() != null) {
   696→                defaultTopicConfigs.put(entry.getKey(), entry.getValue().toString());
   697→            }
   698→        }
   699→        final long windowChangeLogAdditionalRetention = config.getLong(StreamsConfig.WINDOW_STORE_CHANGE_LOG_ADDITIONAL_RETENTION_MS_CONFIG);
   700→
   701→        final Map<String, StreamsRebalanceData.Subtopology> subtopologies = new HashMap<>();
   702→        final Collection<Set<String>> copartitionGroups = internalTopologyBuilder.copartitionGroups();
   703→
   704→        final Set<String> allRepartitionSourceTopics = internalTopologyBuilder.subtopologyToTopicsInfo().values().stream()
   705→            .flatMap(t -> t.repartitionSourceTopics.keySet().stream())
   706→            .collect(Collectors.toSet());
   707→
   708→        for (final Map.Entry<TopologyMetadata.Subtopology, InternalTopologyBuilder.TopicsInfo> topicsInfoEntry : internalTopologyBuilder.subtopologyToTopicsInfo()
   709→            .entrySet()) {
   710→
   711→            final HashSet<String> allSourceTopics = new HashSet<>(topicsInfoEntry.getValue().sourceTopics);
   712→            topicsInfoEntry.getValue().repartitionSourceTopics.forEach(
   713→                (repartitionSourceTopic, repartitionTopicInfo) -> {
   714→                    allSourceTopics.add(repartitionSourceTopic);
   715→                });
   716→
   717→            final Set<String> sourceTopics = topicsInfoEntry.getValue().sourceTopics.stream()
   718→                .filter(topic -> !topicsInfoEntry.getValue().repartitionSourceTopics.containsKey(topic))
   719→                .collect(Collectors.toSet());
   720→            final Set<String> repartitionSinkTopics = topicsInfoEntry.getValue().sinkTopics.stream()
   721→                .filter(allRepartitionSourceTopics::contains)
   722→                .collect(Collectors.toSet());
   723→            final Map<String, StreamsRebalanceData.TopicInfo> repartitionSourceTopics = topicsInfoEntry.getValue().repartitionSourceTopics.entrySet()
   724→                .stream()
   725→                .collect(Collectors.toMap(Map.Entry::getKey, e ->
   726→                    new StreamsRebalanceData.TopicInfo(e.getValue().numberOfPartitions(),
   727→                        Optional.of(config.getInt(StreamsConfig.REPLICATION_FACTOR_CONFIG).shortValue()),
   728→                        e.getValue().properties(defaultTopicConfigs, windowChangeLogAdditionalRetention))));
   729→            final Map<String, StreamsRebalanceData.TopicInfo> stateChangelogTopics = topicsInfoEntry.getValue().stateChangelogTopics.entrySet()
   730→                .stream()
   731→                .collect(Collectors.toMap(Map.Entry::getKey, e ->
   732→                    new StreamsRebalanceData.TopicInfo(e.getValue().numberOfPartitions(),
   733→                        Optional.of(config.getInt(StreamsConfig.REPLICATION_FACTOR_CONFIG).shortValue()),
   734→                        e.getValue().properties(defaultTopicConfigs, windowChangeLogAdditionalRetention))));
   735→
   736→            subtopologies.put(
   737→                String.valueOf(topicsInfoEntry.getKey().nodeGroupId),
   738→                new StreamsRebalanceData.Subtopology(
   739→                    sourceTopics,
   740→                    repartitionSinkTopics,
   741→                    repartitionSourceTopics,
   742→                    stateChangelogTopics,
   743→                    copartitionGroups.stream().filter(allSourceTopics::containsAll).collect(Collectors.toList())
   744→                )
   745→            );
   746→        }
   747→
   748→        if (subtopologies.values().stream().mapToInt(x -> x.copartitionGroups().size()).sum()
   749→            != copartitionGroups.size()) {
   750→            throw new IllegalStateException(
   751→                "Not all copartition groups were converted to broker topology");
   752→        }
   753→
   754→        return subtopologies;
   755→    }

   885→    /**
   886→     * Execute the stream processors
   887→     *
   888→     * @throws KafkaException   for any Kafka-related exceptions
   889→     * @throws RuntimeException for any other non-Kafka exceptions
   890→     */
   891→    @Override
   892→    public void run() {
   893→        log.info("Starting");
   894→        if (setState(State.STARTING) == null) {
   895→            log.info("StreamThread already shutdown. Not running");
   896→            return;
   897→        }
   898→        boolean cleanRun = false;
   899→        try {
   900→            taskManager.init();
   901→            initLatencyWindowsIfNeeded(time.milliseconds());
   902→            cleanRun = runLoop();
   903→        } catch (final Throwable e) {
   904→            failedStreamThreadSensor.record();
   905→            leaveGroupRequested.set(org.apache.kafka.streams.CloseOptions.GroupMembershipOperation.LEAVE_GROUP);
   906→            streamsUncaughtExceptionHandler.accept(e, false);
   907→            // Note: the above call currently rethrows the exception, so nothing below this line will be executed
   908→        } finally {
   909→            completeShutdown(cleanRun);
   910→        }
   911→    }

   913→    /**
   914→     * Main event loop for polling, and processing records through topologies.
   915→     *
   916→     * @throws IllegalStateException If store gets registered after initialized is already finished
   917→     * @throws StreamsException      if the store's change log does not contain the partition
   918→     */
   919→    boolean runLoop() {
   920→        subscribeConsumer();
   921→
   922→        // if the thread is still in the middle of a rebalance, we should keep polling
   923→        // until the rebalance is completed before we close and commit the tasks
   924→        while (isRunning() || taskManager.rebalanceInProgress()) {
   925→            try {
   926→                checkForTopologyUpdates();
   927→                // If we received the shutdown signal while waiting for a topology to be added, we can
   928→                // stop polling regardless of the rebalance status since we know there are no tasks left
   929→                if (!isRunning() && topologyMetadata.isEmpty()) {
   930→                    log.info("Shutting down thread with empty topology.");
   931→                    break;
   932→                }
   933→
   934→                maybeSendShutdown();
   935→                final long size = cacheResizeSize.getAndSet(-1L);
   936→                if (size != -1L) {
   937→                    cacheResizer.accept(size);
   938→                }
   939→                if (processingThreadsEnabled) {
   940→                    runOnceWithProcessingThreads();
   941→                } else {
   942→                    runOnceWithoutProcessingThreads();
   943→                }
   944→
   945→                maybeGetClientInstanceIds();
   946→
   947→                // Check for a scheduled rebalance but don't trigger it until the current rebalance is done
   948→                if (!taskManager.rebalanceInProgress() && nextProbingRebalanceMs.get() < time.milliseconds()) {
   949→                    log.info("Triggering the followup rebalance scheduled for {}.", Utils.toLogDateTimeFormat(nextProbingRebalanceMs.get()));
   950→                    mainConsumer.enforceRebalance("triggered followup rebalance scheduled for " + nextProbingRebalanceMs.get());
   951→                    nextProbingRebalanceMs.set(Long.MAX_VALUE);
   952→                }
   953→            } catch (final TaskCorruptedException e) {
   954→                log.warn("Detected the states of tasks " + e.corruptedTasks() + " are corrupted. " +
   955→                         "Will close the task as dirty and re-create and bootstrap from scratch.", e);
   956→                try {
   957→                    // check if any active task got corrupted. We will trigger a rebalance in that case.
   958→                    // once the task corruptions have been handled
   959→                    final boolean enforceRebalance = taskManager.handleCorruption(e.corruptedTasks());
   960→                    if (enforceRebalance && eosEnabled) {
   961→                        log.info("Active task(s) got corrupted. Triggering a rebalance.");
   962→                        mainConsumer.enforceRebalance("Active tasks corrupted");
   963→                    }
   964→                } catch (final TaskMigratedException taskMigrated) {
   965→                    handleTaskMigrated(taskMigrated);
   966→                }
   967→            } catch (final TaskMigratedException e) {
   968→                handleTaskMigrated(e);
   969→            } catch (final UnsupportedVersionException e) {
   970→                final String errorMessage = e.getMessage();
   971→                if (errorMessage != null &&
   972→                    errorMessage.startsWith("Broker unexpectedly doesn't support requireStable flag on version ")) {
   973→
   974→                    log.error("Shutting down because the Kafka cluster seems to be on a too old version. " +
   975→                              "Setting {}=\"{}\" requires broker version 2.5 or higher.",
   976→                          StreamsConfig.PROCESSING_GUARANTEE_CONFIG,
   977→                          StreamsConfig.EXACTLY_ONCE_V2);
   978→                }
   979→                failedStreamThreadSensor.record();
   980→                this.streamsUncaughtExceptionHandler.accept(new StreamsException(e), false);
   981→                return false;
   982→            } catch (final StreamsException e) {
   983→                throw e;
   984→            } catch (final Exception e) {
   985→                throw new StreamsException(e);
   986→            }
   987→        }
   988→        return true;
   989→    }

   991→    // visible for testing
   992→    void maybeGetClientInstanceIds() {
   993→        // we pass in a timeout of zero into each `clientInstanceId()` call
   994→        // to just trigger the "get instance id" background RPC;
   995→        // we don't want to block the stream thread that can do useful work in the meantime
   996→
   997→        if (fetchDeadlineClientInstanceId != -1) {
   998→            if (!mainConsumerInstanceIdFuture.isDone()) {
   999→                if (fetchDeadlineClientInstanceId >= time.milliseconds()) {
  1000→                    try {
  1001→                        mainConsumerInstanceIdFuture.complete(mainConsumer.clientInstanceId(Duration.ZERO));
  1002→                    } catch (final IllegalStateException disabledError) {
  1003→                        // if telemetry is disabled on a client, we swallow the error,
  1004→                        // to allow returning a partial result for all other clients
  1005→                        mainConsumerInstanceIdFuture.complete(null);
  1006→                    } catch (final TimeoutException swallow) {
  1007→                        // swallow
  1008→                    } catch (final Exception error) {
  1009→                        mainConsumerInstanceIdFuture.completeExceptionally(error);
  1010→                    }
  1011→                } else {
  1012→                    mainConsumerInstanceIdFuture.completeExceptionally(
  1013→                        new TimeoutException("Could not retrieve main consumer client instance id.")
  1014→                    );
  1015→                }
  1016→            }
  1017→
  1018→            if (!producerInstanceIdFuture.isDone()) {
  1019→                if (fetchDeadlineClientInstanceId >= time.milliseconds()) {
  1020→                    try {
  1021→                        producerInstanceIdFuture.complete(
  1022→                            taskManager.streamsProducer().kafkaProducer().clientInstanceId(Duration.ZERO)
  1023→                        );
  1024→                    } catch (final IllegalStateException disabledError) {
  1025→                        // if telemetry is disabled on a client, we swallow the error,
  1026→                        // to allow returning a partial result for all other clients
  1027→                        producerInstanceIdFuture.complete(null);
  1028→                    } catch (final TimeoutException swallow) {
  1029→                        // swallow
  1030→                    } catch (final Exception error) {
  1031→                        producerInstanceIdFuture.completeExceptionally(error);
  1032→                    }
  1033→                } else {
  1034→                    producerInstanceIdFuture.completeExceptionally(
  1035→                        new TimeoutException("Could not retrieve thread producer client instance id.")
  1036→                    );
  1037→                }
  1038→            }
  1039→
  1040→            if (mainConsumerInstanceIdFuture.isDone() && producerInstanceIdFuture.isDone()) {
  1041→                fetchDeadlineClientInstanceId = -1L;
  1042→            }
  1043→        }
  1044→    }

  1069→    public boolean waitOnThreadState(final StreamThread.State targetState, final long timeoutMs) {
  1070→        final long begin = time.milliseconds();
  1071→        synchronized (stateLock) {
  1072→            boolean interrupted = false;
  1073→            long elapsedMs = 0L;
  1074→            try {
  1075→                while (state != targetState) {
  1076→                    if (timeoutMs >= elapsedMs) {
  1077→                        final long remainingMs = timeoutMs - elapsedMs;
  1078→                        try {
  1079→                            stateLock.wait(remainingMs);
  1080→                        } catch (final InterruptedException e) {
  1081→                            interrupted = true;
  1082→                        }
  1083→                    } else {
  1084→                        log.debug("Cannot transit to {} within {}ms", targetState, timeoutMs);
  1085→                        return false;
  1086→                    }
  1087→                    elapsedMs = time.milliseconds() - begin;
  1088→                }
  1089→                return true;
  1090→            } finally {
  1091→                // Make sure to restore the interruption status before returning.
  1092→                // We do not always own the current thread that executes this method, i.e., we do not know the
  1093→                // interruption policy of the thread. The least we can do is restore the interruption status before
  1094→                // the current thread exits this method.
  1095→                if (interrupted) {
  1096→                    Thread.currentThread().interrupt();
  1097→                }
  1098→            }
  1099→        }
  1100→    }

  1151→    /**
  1152→     * One iteration of a thread includes the following steps:
  1153→     * <p>
  1154→     * 1. poll records from main consumer and add to buffer;
  1155→     * 2. restore from restore consumer and update standby tasks if necessary;
  1156→     * 3. process active tasks from the buffers;
  1157→     * 4. punctuate active tasks if necessary;
  1158→     * 5. commit all tasks if necessary;
  1159→     *
  1160→     * <p> Among them, step 3/4/5 is done in batches in which we try to process as much as possible while trying to
  1161→     * stop iteration to call the next iteration when it's close to the next main consumer's poll deadline
  1162→     *
  1163→     * @throws IllegalStateException If store gets registered after initialized is already finished
  1164→     * @throws StreamsException      If the store's change log does not contain the partition
  1165→     * @throws TaskMigratedException If another thread wrote to the changelog topic that is currently restored
  1166→     *                               or if committing offsets failed (non-EOS)
  1167→     *                               or if the task producer got fenced (EOS)
  1168→     */
  1169→    // Visible for testing
  1170→    void runOnceWithoutProcessingThreads() {
  1171→        final long startMs = time.milliseconds();
  1172→        now = startMs;
  1173→
  1174→        final long pollLatency;
  1175→        taskManager.resumePollingForPartitionsWithAvailableSpace();
  1176→        pollLatency = pollPhase();
  1177→        totalPolledSinceLastSummary += 1;
  1178→
  1179→        if (streamsRebalanceData.isPresent()) {
  1180→            // Always handle status codes (e.g., MISSING_SOURCE_TOPICS, INCORRECTLY_PARTITIONED_TOPICS)
  1181→            // regardless of streamsGroupReady, as these may throw exceptions that need to be handled.
  1182→            handleStreamsRebalanceData();
  1183→
  1184→            if (!streamsGroupReady) {
  1185→                return;
  1186→            }
  1187→        }
  1188→
  1189→        // Shutdown hook could potentially be triggered and transit the thread state to PENDING_SHUTDOWN during #pollRequests().
  1190→        // The task manager internal states could be uninitialized if the state transition happens during #onPartitionsAssigned().
  1191→        // Should only proceed when the thread is still running after #pollRequests(), because no external state mutation
  1192→        // could affect the task manager state beyond this point within #runOnce().
  1193→        if (!isRunning()) {
  1194→            log.debug("Thread state is already {}, skipping the run once call after poll request", state);
  1195→            return;
  1196→        }
  1197→
  1198→        // TODO: we should record the restore latency and its relative time spent ratio after
  1199→        //       we figure out how to move this method out of the stream thread
  1200→        advanceNowAndComputeLatency();
  1201→
  1202→        int totalProcessed = 0;
  1203→        long totalCommitLatency = 0L;
  1204→        long totalProcessLatency = 0L;
  1205→        long totalPunctuateLatency = 0L;
  1206→        if (isStartingRunningOrPartitionAssigned()) {
  1207→
  1208→            taskManager.updateLags();
  1209→
  1210→            /*
  1211→             * Within an iteration, after processing up to N (N initialized as 1 upon start up) records for each applicable tasks, check the current time:
  1212→             *  1. If it is time to punctuate, do it;
  1213→             *  2. If it is time to commit, do it, this should be after 1) since punctuate may trigger commit;
  1214→             *  3. If there's no records processed, end the current iteration immediately;
  1215→             *  4. If we are close to consumer's next poll deadline, end the current iteration immediately;
  1216→             *  5. If any of 1), 2) and 4) happens, half N for next iteration;
  1217→             *  6. Otherwise, increment N.
  1218→             */
  1219→            do {
  1220→
  1221→                checkStateUpdater();
  1222→
  1223→                log.debug("Processing tasks with {} iterations.", numIterations);
  1224→                final int processed = taskManager.process(numIterations, time);
  1225→                final long processLatency = advanceNowAndComputeLatency();
  1226→                totalProcessLatency += processLatency;
  1227→                if (processed > 0) {
  1228→                    // It makes no difference to the outcome of these metrics when we record "0",
  1229→                    // so we can just avoid the method call when we didn't process anything.
  1230→                    processRateSensor.record(processed, now);
  1231→
  1232→                    // This metric is scaled to represent the _average_ processing time of _each_
  1233→                    // task. Note, it's hard to interpret this as defined; the per-task process-ratio
  1234→                    // as well as total time ratio spent on processing compared with polling / committing etc
  1235→                    // are reported on other metrics.
  1236→                    processLatencySensor.record(processLatency / (double) processed, now);
  1237→
  1238→                    totalProcessed += processed;
  1239→                    totalRecordsProcessedSinceLastSummary += processed;
  1240→                }
  1241→
  1242→                log.debug("Processed {} records with {} iterations; invoking punctuators if necessary",
  1243→                          processed,
  1244→                          numIterations);
  1245→
  1246→                final int punctuated = taskManager.punctuate();
  1247→                totalPunctuatorsSinceLastSummary += punctuated;
  1248→                final long punctuateLatency = advanceNowAndComputeLatency();
  1249→                totalPunctuateLatency += punctuateLatency;
  1250→                if (punctuated > 0) {
  1251→                    punctuateSensor.record(punctuateLatency / (double) punctuated, now);
  1252→                }
  1253→
  1254→                log.debug("{} punctuators ran.", punctuated);
  1255→
  1256→                final long beforeCommitMs = now;
  1257→                final int committed = maybeCommit();
  1258→                final long commitLatency = Math.max(now - beforeCommitMs, 0);
  1259→                totalCommitLatency += commitLatency;
  1260→                if (committed > 0) {
  1261→                    totalCommittedSinceLastSummary += committed;
  1262→                    commitSensor.record(commitLatency / (double) committed, now);
  1263→
  1264→                    if (log.isDebugEnabled()) {
  1265→                        log.debug("Committed all active tasks {} and standby tasks {} in {}ms",
  1266→                            taskManager.activeRunningTaskIds(), taskManager.standbyTaskIds(), commitLatency);
  1267→                    }
  1268→                }
  1269→
  1270→                if (processed == 0) {
  1271→                    // if there are no records to be processed, exit after punctuate / commit
  1272→                    break;
  1273→                } else if (Math.max(now - lastPollMs, 0) > maxPollTimeMs / 2) {
  1274→                    numIterations = numIterations > 1 ? numIterations / 2 : numIterations;
  1275→                    break;
  1276→                } else if (punctuated > 0 || committed > 0) {
  1277→                    numIterations = numIterations > 1 ? numIterations / 2 : numIterations;
  1278→                } else {
  1279→                    numIterations++;
  1280→                }
  1281→            } while (true);
  1282→
  1283→            // we record the ratio out of the while loop so that the accumulated latency spans over
  1284→            // multiple iterations with reasonably large max.num.records and hence is less vulnerable to outliers
  1285→            taskManager.recordTaskProcessRatio(totalProcessLatency, now);
  1286→        }
  1287→
  1288→        now = time.milliseconds();
  1289→        final long runOnceLatency = now - startMs;
  1290→
  1291→        recordWindowedSum(now, pollLatency, totalCommitLatency, totalProcessLatency, totalPunctuateLatency, runOnceLatency);
  1292→        recordRatio(now, pollLatencyWindowedSum, pollRatioSensor);
  1293→        recordRatio(now, totalCommitLatencyWindowedSum, commitRatioSensor);
  1294→        recordRatio(now, processLatencyWindowedSum, processRatioSensor);
  1295→        recordRatio(now, punctuateLatencyWindowedSum, punctuateRatioSensor);
  1296→        processRecordsSensor.record(totalProcessed, now);
  1297→
  1298→        final long timeSinceLastLog = now - lastLogSummaryMs;
  1299→        if (logSummaryIntervalMs > 0 && timeSinceLastLog > logSummaryIntervalMs) {
  1300→            log.info("Processed {} total records, ran {} punctuators, polled {} times and committed {} total tasks since the last update {}ms ago",
  1301→                 totalRecordsProcessedSinceLastSummary, totalPunctuatorsSinceLastSummary, totalPolledSinceLastSummary, totalCommittedSinceLastSummary, timeSinceLastLog);
  1302→
  1303→            totalRecordsProcessedSinceLastSummary = 0L;
  1304→            totalPunctuatorsSinceLastSummary = 0L;
  1305→            totalPolledSinceLastSummary = 0L;
  1306→            totalCommittedSinceLastSummary = 0L;
  1307→            lastLogSummaryMs = now;
  1308→        }
  1309→    }

  1417→    private long pollPhase() {
  1418→        final ConsumerRecords<byte[], byte[]> records;
  1419→        log.debug("Invoking poll on main Consumer");
  1420→
  1421→        if (state == State.PARTITIONS_REVOKED) {
  1422→            // try to fetch some records with zero poll millis to unblock
  1423→            // other useful work while waiting for the join response
  1424→            records = pollRequests(Duration.ZERO);
  1425→        } else if (state == State.RUNNING || state == State.STARTING || state == State.PARTITIONS_ASSIGNED) {
  1426→            // try to fetch some records with normal poll time
  1427→            // in order to get long polling
  1428→            records = pollRequests(pollTime);
  1429→        } else if (state == State.PENDING_SHUTDOWN) {
  1430→            // we are only here because there's rebalance in progress,
  1431→            // just poll with zero to complete it
  1432→            records = pollRequests(Duration.ZERO);
  1433→        } else {
  1434→            // any other state should not happen
  1435→            log.error("Unexpected state {} during normal iteration", state);
  1436→            throw new StreamsException(logPrefix + "Unexpected state " + state + " during normal iteration");
  1437→        }
  1438→
  1439→        final long pollLatency = advanceNowAndComputeLatency();
  1440→
  1441→        final int numRecords = records.count();
  1442→
  1443→        for (final TopicPartition topicPartition: records.partitions()) {
  1444→            records
  1445→                .records(topicPartition)
  1446→                .stream()
  1447→                .max(Comparator.comparing(ConsumerRecord::offset))
  1448→                .ifPresent(t -> taskManager.updateTaskEndMetadata(topicPartition, t.offset()));
  1449→        }
  1450→
  1451→        log.debug("Main Consumer poll completed in {} ms and fetched {} records from partitions {}",
  1452→            pollLatency, numRecords, records.partitions());
  1453→
  1454→        pollSensor.record(pollLatency, now);
  1455→
  1456→        if (!records.isEmpty()) {
  1457→            pollRecordsSensor.record(numRecords, now);
  1458→            taskManager.addRecordsToTasks(records);
  1459→        }
  1460→        if (!records.nextOffsets().isEmpty()) {
  1461→            taskManager.updateNextOffsets(records.nextOffsets());
  1462→        }
  1463→
  1464→        while (!nonFatalExceptionsToHandle.isEmpty()) {
  1465→            streamsUncaughtExceptionHandler.accept(nonFatalExceptionsToHandle.poll(), true);
  1466→        }
  1467→        return pollLatency;
  1468→    }

  1503→    public void handleStreamsRebalanceData() {
  1504→        if (streamsRebalanceData.isPresent()) {
  1505→            boolean hasMissingSourceTopics = false;
  1506→            String missingTopicsDetail = null;
  1507→
  1508→            for (final StreamsGroupHeartbeatResponseData.Status status : streamsRebalanceData.get().statuses()) {
  1509→                if (status.statusCode() == StreamsGroupHeartbeatResponse.Status.SHUTDOWN_APPLICATION.code()) {
  1510→                    shutdownErrorHook.run();
  1511→                } else if (status.statusCode() == StreamsGroupHeartbeatResponse.Status.MISSING_SOURCE_TOPICS.code()) {
  1512→                    hasMissingSourceTopics = true;
  1513→                    missingTopicsDetail = status.statusDetail();
  1514→                } else if (status.statusCode() == StreamsGroupHeartbeatResponse.Status.INCORRECTLY_PARTITIONED_TOPICS.code()) {
  1515→                    final String errorMsg = status.statusDetail();
  1516→                    log.error(errorMsg);
  1517→                    throw new TopologyException(errorMsg);
  1518→                }
  1519→            }
  1520→
  1521→            if (hasMissingSourceTopics) {
  1522→                handleMissingSourceTopicsWithTimeout(missingTopicsDetail);
  1523→            } else {
  1524→                // Reset timeout tracking when no missing source topics are reported
  1525→                topicsReadyTimer = null;
  1526→            }
  1527→
  1528→            final Map<StreamsRebalanceData.HostInfo, StreamsRebalanceData.EndpointPartitions> partitionsByEndpoint =
  1529→                    streamsRebalanceData.get().partitionsByHost();
  1530→            final Map<HostInfo, Set<TopicPartition>> activeHostInfoMap = new HashMap<>();
  1531→            final Map<HostInfo, Set<TopicPartition>> standbyHostInfoMap = new HashMap<>();
  1532→
  1533→            partitionsByEndpoint.forEach((hostInfo, endpointPartitions) -> {
  1534→                activeHostInfoMap.put(new HostInfo(hostInfo.host(), hostInfo.port()), new HashSet<>(endpointPartitions.activePartitions()));
  1535→                standbyHostInfoMap.put(new HostInfo(hostInfo.host(), hostInfo.port()), new HashSet<>(endpointPartitions.standbyPartitions()));
  1536→            });
  1537→            streamsMetadataState.onChange(
  1538→                    activeHostInfoMap,
  1539→                    standbyHostInfoMap,
  1540→                    getTopicPartitionInfo(activeHostInfoMap)
  1541→            );
  1542→        }
  1543→    }

  1545→    private void handleMissingSourceTopicsWithTimeout(final String missingTopicsDetail) {
  1546→        // Use 2 * heartbeatIntervalMs as the timeout ensures at least one heartbeat is sent before raising the exception
  1547→        final int heartbeatIntervalMs = streamsRebalanceData.get().heartbeatIntervalMs();
  1548→        final long timeoutMs = 2L * heartbeatIntervalMs;
  1549→
  1550→        // Start timeout tracking on first encounter with missing topics
  1551→        if (topicsReadyTimer == null) {
  1552→            topicsReadyTimer = time.timer(timeoutMs);
  1553→            log.info("Missing source topics detected: {}. Will wait up to {}ms before failing.",
  1554→                missingTopicsDetail, timeoutMs);
  1555→        } else {
  1556→            topicsReadyTimer.update();
  1557→        }
  1558→
  1559→        if (topicsReadyTimer.isExpired()) {
  1560→            final long elapsedTime = topicsReadyTimer.elapsedMs();
  1561→            final String errorMsg = String.format("Missing source topics: %s. Timeout exceeded after %dms.",
  1562→                missingTopicsDetail, elapsedTime);
  1563→            log.error(errorMsg);
  1564→

  1592→    private void resetOffsets(final Set<TopicPartition> partitions, final Exception cause) {
  1593→        final Set<String> loggedTopics = new HashSet<>();
  1594→        final Set<TopicPartition> seekToBeginning = new HashSet<>();
  1595→        final Set<TopicPartition> seekToEnd = new HashSet<>();
  1596→        final Map<TopicPartition, Duration> seekByDuration = new HashMap<>();
  1597→        final Set<TopicPartition> notReset = new HashSet<>();
  1598→
  1599→        for (final TopicPartition partition : partitions) {
  1600→            final Optional<AutoOffsetResetStrategy> offsetResetStrategy = topologyMetadata.offsetResetStrategy(partition.topic());
  1601→
  1602→            // TODO
  1603→            // This may be null if the task we are currently processing was part of a named topology that was just removed.
  1604→            // After named topologies are removed, we can update `topologyMetadata.offsetResetStrateg()` so it
  1605→            // will not return null any longer, and we can remove this check
  1606→            if (offsetResetStrategy != null) {
  1607→                if (offsetResetStrategy.isPresent()) {
  1608→                    final AutoOffsetResetStrategy resetPolicy = offsetResetStrategy.get();
  1609→
  1610→                    if (resetPolicy == AutoOffsetResetStrategy.NONE) {
  1611→                        notReset.add(partition);
  1612→                    } else if (resetPolicy == AutoOffsetResetStrategy.EARLIEST) {
  1613→                        addToResetList(
  1614→                            partition,
  1615→                            seekToBeginning,
  1616→                            "Setting topic '{}' to consume from 'earliest' offset",
  1617→                            loggedTopics
  1618→                        );
  1619→                    } else if (resetPolicy == AutoOffsetResetStrategy.LATEST) {
  1620→                        addToResetList(
  1621→                            partition,
  1622→                            seekToEnd,
  1623→                            "Setting topic '{}' to consume from 'latest' offset",
  1624→                            loggedTopics
  1625→                        );
  1626→                    } else if (resetPolicy.type() == AutoOffsetResetStrategy.StrategyType.BY_DURATION) {
  1627→                        addToResetList(
  1628→                            partition,
  1629→                            seekByDuration,
  1630→                            resetPolicy.duration().get(),
  1631→                            "Setting topic '{}' to consume from 'by_duration:{}'",
  1632→                            resetPolicy.duration().get().toString(),
  1633→                            loggedTopics
  1634→                        );
  1635→                    } else {
  1636→                        throw new IllegalStateException("Unknown reset policy " + resetPolicy);
  1637→                    }
  1638→                } else {
  1639→                    final AutoOffsetResetStrategy resetPolicy = AutoOffsetResetStrategy.fromString(originalReset);
  1640→
  1641→                    if (resetPolicy == AutoOffsetResetStrategy.NONE) {
  1642→                        notReset.add(partition);
  1643→                    } else if (resetPolicy == AutoOffsetResetStrategy.EARLIEST) {
  1644→                        addToResetList(
  1645→                            partition,
  1646→                            seekToBeginning,
  1647→                            "No custom setting defined for topic '{}' using original config 'earliest' for offset reset",
  1648→                            loggedTopics
  1649→                        );
  1650→                    } else if (resetPolicy == AutoOffsetResetStrategy.LATEST) {
  1651→                        addToResetList(
  1652→                            partition,
  1653→                            seekToEnd,
  1654→                            "No custom setting defined for topic '{}' using original config 'latest' for offset reset",
  1655→                            loggedTopics
  1656→                        );
  1657→                    } else if (resetPolicy.type() == AutoOffsetResetStrategy.StrategyType.BY_DURATION) {
  1658→                        addToResetList(
  1659→                            partition,
  1660→                            seekByDuration,
  1661→                            resetPolicy.duration().get(),
  1662→                            "No custom setting defined for topic '{}' using original config 'by_duration:{}' for offset reset",
  1663→                            resetPolicy.duration().get().toString(),
  1664→                            loggedTopics
  1665→                        );
  1666→                    } else {
  1667→                        throw new IllegalStateException("Unknown reset policy " + resetPolicy);
  1668→                    }
  1669→                }
  1670→            }
  1671→        }
  1672→
  1673→        if (notReset.isEmpty()) {
  1674→            if (!seekToBeginning.isEmpty()) {
  1675→                mainConsumer.seekToBeginning(seekToBeginning);
  1676→            }
  1677→
  1678→            if (!seekToEnd.isEmpty()) {
  1679→                mainConsumer.seekToEnd(seekToEnd);
  1680→            }
  1681→
  1682→            if (!seekByDuration.isEmpty()) {
  1683→                final long nowMs = time.milliseconds();
  1684→                final Map<TopicPartition, Long> seekToTimestamps = seekByDuration.entrySet().stream()
  1685→                    .map(e -> {
  1686→                        long seekMs = nowMs - e.getValue().toMillis();
  1687→                        if (seekMs < 0L) {
  1688→                            log.debug("Cannot reset offset to negative timestamp {} for partition {}. Seeking to timestamp 0 instead.", seekMs, e.getKey());
  1689→                            seekMs = 0L;
  1690→                        }
  1691→                        return Map.entry(e.getKey(), seekMs);
  1692→                    })
  1693→                    .collect(HashMap::new, (m, e) -> m.put(e.getKey(), e.getValue()), Map::putAll);
  1694→
  1695→                try {
  1696→                    for (final Map.Entry<TopicPartition, OffsetAndTimestamp> partitionAndOffset : mainConsumer.offsetsForTimes(seekToTimestamps).entrySet()) {
  1697→                        final TopicPartition partition = partitionAndOffset.getKey();
  1698→                        final OffsetAndTimestamp seekOffset = partitionAndOffset.getValue();
  1699→                        if (seekOffset != null) {
  1700→                            mainConsumer.seek(partition, new OffsetAndMetadata(seekOffset.offset()));
  1701→                        } else {
  1702→                            log.debug(
  1703→                                "Cannot reset offset to non-existing timestamp {} (larger than timestamp of last record)" +
  1704→                                    " for partition {}. Seeking to end instead.",
  1705→                                seekToTimestamps.get(partition),
  1706→                                partition
  1707→                            );
  1708→                            mainConsumer.seekToEnd(Collections.singleton(partitionAndOffset.getKey()));
  1709→                        }
  1710→                    }
  1711→                } catch (final TimeoutException timeoutException) {
  1712→                    taskManager.maybeInitTaskTimeoutsOrThrow(seekByDuration.keySet(), timeoutException, now);
  1713→                    log.debug(
  1714→                        String.format(
  1715→                            "Could not reset offset for %s due to the following exception; will retry.",
  1716→                            seekByDuration.keySet()),
  1717→                        timeoutException
  1718→                    );
  1719→                }
  1720→            }
  1721→        } else {
  1722→            final String notResetString =
  1723→                notReset.stream()
  1724→                        .map(TopicPartition::topic)
  1725→                        .distinct()
  1726→                        .collect(Collectors.joining(","));
  1727→
  1728→            final String format = String.format(
  1729→                "No valid committed offset found for input [%s] and no valid reset policy configured." +
  1730→                    " You need to set configuration parameter \"auto.offset.reset\" or specify a topic specific reset " +
  1731→                    "policy via StreamsBuilder#stream(..., Consumed.with(Topology.AutoOffsetReset)) or " +
  1732→                    "StreamsBuilder#table(..., Consumed.with(Topology.AutoOffsetReset))",
  1733→                notResetString
  1734→            );
  1735→
  1736→            if (cause == null) {
  1737→                throw new StreamsException(format);
  1738→            } else {
  1739→                throw new StreamsException(format, cause);
  1740→            }
  1741→        }
  1742→    }

  1790→        final int committed;
  1791→        if (now - lastCommitMs > commitTimeMs) {
  1792→            if (log.isDebugEnabled()) {
  1793→                log.debug("Committing all active tasks {} and standby tasks {} since {}ms has elapsed (commit interval is {}ms)",
  1794→                          taskManager.activeRunningTaskIds(), taskManager.standbyTaskIds(), now - lastCommitMs, commitTimeMs);

  1797→            committed = taskManager.commit(
  1798→                taskManager.allRunningTasks()
  1799→                    .values()
  1800→                    .stream()
  1801→                    .filter(t -> t.state() == Task.State.RUNNING || t.state() == Task.State.RESTORING)
  1802→                    .collect(Collectors.toSet())
  1803→            );
  1804→
  1805→            if (committed == -1) {
  1806→                log.debug("Unable to commit as we are in the middle of a rebalance, will try again when it completes.");
  1807→            } else {
  1808→                now = time.milliseconds();
  1809→                lastCommitMs = now;
  1810→            }
  1811→        } else {
  1812→            committed = taskManager.maybeCommitActiveTasksPerUserRequested();
  1813→        }
  1814→
  1815→        if ((now - lastPurgeMs) > purgeTimeMs) {
  1816→            // try to purge the committed records for repartition topics if possible
  1817→            taskManager.maybePurgeCommittedRecords();
  1818→            lastPurgeMs = now;
  1819→        }
  1820→
  1821→        return committed;

  1837→    /**
  1838→     * Shutdown this stream thread.
  1839→     * <p>
  1840→     * Note that there is nothing to prevent this function from being called multiple times
  1841→     * (e.g., in testing), hence the state is set only the first time
  1842→     *
  1843→     * @param operation the group membership operation to apply on shutdown. Must be one of LEAVE_GROUP or REMAIN_IN_GROUP.

  1845→    public void shutdown(final org.apache.kafka.streams.CloseOptions.GroupMembershipOperation operation) {
  1846→        log.info("Informed to shut down");
  1847→        final State oldState = setState(State.PENDING_SHUTDOWN);
  1848→        leaveGroupRequested.set(operation);
  1849→        if (oldState == State.CREATED) {

  1855→    private void completeShutdown(final boolean cleanRun) {
  1856→        // set the state to pending shutdown first as it may be called due to error;
  1857→        // its state may already be PENDING_SHUTDOWN so it will return false but we
  1858→        // intentionally do not check the returned flag
  1859→        setState(State.PENDING_SHUTDOWN);
  1860→
  1861→        log.info("Shutting down {}", cleanRun ? "clean" : "unclean");
  1862→
  1863→        mainConsumerInstanceIdFuture.complete(null);
  1864→
  1865→        try {
  1866→            taskManager.shutdown(cleanRun);
  1867→        } catch (final Throwable e) {
  1868→            log.error("Failed to close task manager due to the following error:", e);
  1869→        }
  1870→        try {
  1871→            topologyMetadata.unregisterThread(threadMetadata.threadName());
  1872→        } catch (final Throwable e) {
  1873→            log.error("Failed to unregister thread due to the following error:", e);
  1874→        }
  1875→        try {
  1876→            changelogReader.clear();
  1877→        } catch (final Throwable e) {
  1878→            log.error("Failed to close changelog reader due to the following error:", e);
  1879→        }
  1880→        try {
  1881→            final GroupMembershipOperation membershipOperation =
  1882→                leaveGroupRequested.get() == org.apache.kafka.streams.CloseOptions.GroupMembershipOperation.LEAVE_GROUP ? LEAVE_GROUP : REMAIN_IN_GROUP;
  1883→            if (membershipOperation == REMAIN_IN_GROUP && streamsRebalanceData.isPresent()) {
  1884→                log.info("The consumer will leave the group since the streams group protocol is used");
  1885→            }
  1886→            mainConsumer.close(CloseOptions.groupMembershipOperation(membershipOperation));
  1887→        } catch (final Throwable e) {
  1888→            log.error("Failed to close consumer due to the following error:", e);
  1889→        }
  1890→        try {
  1891→            // restore consumer isn't part of a consumer group so we use REMAIN_IN_GROUP to skip any leaveGroup checks
  1892→            restoreConsumer.close();
  1893→        } catch (final Throwable e) {
  1894→            log.error("Failed to close restore consumer due to the following error:", e);
  1895→        }
  1896→        streamsMetrics.removeAllThreadLevelSensors(getName());
  1897→        streamsMetrics.removeAllThreadLevelMetrics(getName());
  1898→
  1899→        setState(State.DEAD);
  1900→
  1901→        log.info("Shutdown complete");
  1902→    }

  2124→    private void recordRatio(final long now, final WindowedSum windowedSum, final Sensor ratioSensor) {
  2125→        final double runOnceLatencyWindow =
  2126→            runOnceLatencyWindowedSum.measure(metricsConfig, now);
  2127→
  2128→        if (runOnceLatencyWindow > 0.0) {
  2129→            final double latencyWindow =
  2130→                windowedSum.measure(metricsConfig, now);
